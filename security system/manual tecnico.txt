#include <Servo.h>

Servo miServo;  // Crear un objeto servo para controlar el servo motor

void setup() {
  miServo.attach(9);  // El servo está conectado al pin digital 9
}

void loop() {
  abrirPuerta();      // Llamar a la función para abrir la puerta
  delay(5000);        // Esperar 5 segundos
  cerrarPuerta();     // Llamar a la función para cerrar la puerta
  delay(5000);        // Esperar 5 segundos
}

void abrirPuerta() {
  miServo.write(0);   // Ajustar el ángulo para abrir la puerta
  delay(1000);        // Mantener la posición durante 1 segundo (ajusta según sea necesario)
}

void cerrarPuerta() {
  miServo.write(320);  // Ajustar el ángulo para cerrar la puerta
  delay(1000);         // Mantener la posición durante 1 segundo (ajusta según sea necesario)
}
